{
  "resources": {
    "azure_ad_saml": {
      "type": "aws/iam/samlProvider",
      "metadata": {
        "displayName": "Azure AD SAML Provider",
        "description": "SAML provider for Azure Active Directory SSO"
      },
      "spec": {
        // The name of the SAML provider to create.
        // This parameter allows a string of characters consisting of upper and 
        // lowercase alphanumeric characters with no spaces. You can also include 
        // any of the following characters: _+=,.@-
        "name": "AzureADSAML",
        
        // An XML document generated by an identity provider (IdP) that supports 
        // SAML 2.0. The document includes the issuer's name, expiration information,
        // and keys that can be used to validate the SAML authentication response 
        // (assertions) that are received from the IdP.
        "samlMetadataDocument": "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<EntityDescriptor xmlns=\"urn:oasis:names:tc:SAML:2.0:metadata\" \n                  xmlns:ds=\"http://www.w3.org/2000/09/xmldsig#\" \n                  entityID=\"https://sts.windows.net/12345678-1234-1234-1234-123456789012/\">\n  <IDPSSODescriptor protocolSupportEnumeration=\"urn:oasis:names:tc:SAML:2.0:protocol\">\n    <KeyDescriptor use=\"signing\">\n      <KeyInfo>\n        <X509Data>\n          <X509Certificate>MIIDBTCCAe2gAwIBAgIQY...</X509Certificate>\n        </X509Data>\n      </KeyInfo>\n    </KeyDescriptor>\n    <SingleLogoutService Binding=\"urn:oasis:names:tc:SAML:2.0:bindings:HTTP-Redirect\" \n                        Location=\"https://login.microsoftonline.com/12345678-1234-1234-1234-123456789012/saml2\"/>\n    <NameIDFormat>urn:oasis:names:tc:SAML:2.0:nameid-format:persistent</NameIDFormat>\n    <SingleSignOnService Binding=\"urn:oasis:names:tc:SAML:2.0:bindings:HTTP-Redirect\" \n                        Location=\"https://login.microsoftonline.com/12345678-1234-1234-1234-123456789012/saml2\"/>\n  </IDPSSODescriptor>\n</EntityDescriptor>",
        
        // Optional: Tags to attach to the SAML provider
        "tags": [
          {
            "key": "Environment",
            "value": "Production"
          },
          {
            "key": "IdentityProvider",
            "value": "AzureAD"
          },
          {
            "key": "CostCenter",
            "value": "IT-Security"
          }
        ]
      }
    }
  }
}
